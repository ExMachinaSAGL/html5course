                font-family: 'Airstream';
                src: url('..//Airstream-webfont.eot'); /* IE 5-8 */

                url('../Airstream-webfont.woff') format('woff'),/* FF 3.6, Chrome 5, IE9 */
                url('../Airstream-webfont.ttf') format('truetype'), /* Opera, Safari */
                url('../Airstream-webfont.svg#font') format('svg'); /* iOS */
            }
